    ```typescript
    import { useState, useEffect } from 'react';

    export function useDebounce<T>(value: T, delay: number = 300): T {
      const [debouncedValue, setDebouncedValue] = useState<T>(value);

      useEffect(() => {
        const timer = setTimeout(() => {
          setDebouncedValue(value);
        }, delay);

        return () => {
          clearTimeout(timer);
        };
      }, [value, delay]);

      return debouncedValue;
    }
    ```

    This hook:
    - Delays updating the returned value until `delay` ms after the last `value` change
    - Cleans up timer on unmount or value change
    - Uses generics to support any type (string, number, object)
    - Default 300ms delay per research findings (reduces API calls by 80%)

    import { supabase } from '@/lib/supabase';
    import { UserManualAccess } from '@/types/supabase';

    export interface AdminUser extends UserManualAccess {
      user_email: string | null;
      granter_email: string | null;
    }

    export function useAdminUsers(search?: string) {
      return useQuery({
        queryKey: ['admin-users', 'search', search || ''],
        queryFn: async () => {
          const token = (await supabase.auth.getSession()).data.session?.access_token;
          if (!token) throw new Error('No auth token');

          const searchParam = search ? `?search=${encodeURIComponent(search)}` : '';
          const response = await fetch(`/api/admin/access/list${searchParam}`, {
            headers: {
              'Authorization': `Bearer ${token}`,
            },
          });

          if (!response.ok) {
            throw new Error('Failed to fetch users');
          }

          const data = await response.json();
          return data.accesses as AdminUser[];
        },
        staleTime: 1000 * 60 * 5, // 5 minutes
      });
    }
    ```

    Key features:
    - Uses useQuery for automatic caching and refetching
    - Query key includes search term for proper cache invalidation
    - Fetches with auth token from Supabase session
    - Passes search parameter to API (server-side filtering)
    - 5-minute stale time reduces unnecessary refetches

    import { useMutation, useQueryClient } from '@tanstack/react-query';
    import { toast } from 'sonner';
    import { supabase } from '@/lib/supabase';
    import { AdminUser } from './useAdminUsers';

    interface GrantAccessInput {
      email: string;
      reason?: string;
      expires_at?: string;
    }

    export function useGrantAccess() {
      const queryClient = useQueryClient();

      return useMutation({
        mutationFn: async (data: GrantAccessInput) => {
          const token = (await supabase.auth.getSession()).data.session?.access_token;
          if (!token) throw new Error('No auth token');

          const response = await fetch('/api/admin/access/grant', {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json',
              'Authorization': `Bearer ${token}`,
            },
            body: JSON.stringify(data),
          });

          if (!response.ok) {
            const error = await response.json();
            throw new Error(error.error || 'Failed to grant access');
          }

          return response.json();
        },

        onMutate: async (newAccess) => {
          await queryClient.cancelQueries({ queryKey: ['admin-users'] });

          const previousUsers = queryClient.getQueryData<AdminUser[]>(['admin-users', 'search', '']);

          queryClient.setQueryData(['admin-users', 'search', ''], (old: AdminUser[] | undefined) => {
            if (!old) return old;
            return [
              {
                id: 'temp-' + Date.now(),
                user_id: 'temp-user',
                granted_by: 'temp-admin',
                granted_at: new Date().toISOString(),
                expires_at: newAccess.expires_at || null,
                reason: newAccess.reason || null,
                is_active: true,
                user_email: newAccess.email,
                granter_email: null,
              },
              ...old,
            ];
          });

          return { previousUsers };
        },

        onError: (err, newAccess, context) => {
          queryClient.setQueryData(['admin-users', 'search', ''], context?.previousUsers);
          toast.error('Erro ao conceder acesso', {
            description: err instanceof Error ? err.message : 'Tente novamente',
          });
        },

        onSettled: () => {
          queryClient.invalidateQueries({ queryKey: ['admin-users'] });
        },

        onSuccess: () => {
          toast.success('Acesso concedido com sucesso');
        },
      });
    }

    export function useRevokeAccess() {
      const queryClient = useQueryClient();

      return useMutation({
        mutationFn: async ({ userId, email }: { userId: string; email: string }) => {
          const token = (await supabase.auth.getSession()).data.session?.access_token;
          if (!token) throw new Error('No auth token');

          const response = await fetch(`/api/admin/access/${userId}`, {
            method: 'DELETE',
            headers: {
              'Authorization': `Bearer ${token}`,
            },
          });

          if (!response.ok) {
            const error = await response.json();
            throw new Error(error.error || 'Failed to revoke access');
          }

          return response.json();
        },

        onMutate: async ({ userId }) => {
          await queryClient.cancelQueries({ queryKey: ['admin-users'] });

          const previousUsers = queryClient.getQueryData<AdminUser[]>(['admin-users', 'search', '']);

          queryClient.setQueryData(['admin-users', 'search', ''], (old: AdminUser[] | undefined) => {
            if (!old) return old;
            return old.map(user =>
              user.user_id === userId
                ? { ...user, is_active: false }
                : user
            );
          });

          return { previousUsers };
        },

        onError: (err, variables, context) => {
          queryClient.setQueryData(['admin-users', 'search', ''], context?.previousUsers);
          toast.error('Erro ao revogar acesso', {
            description: err instanceof Error ? err.message : 'Tente novamente',
          });
        },

        onSettled: () => {
          queryClient.invalidateQueries({ queryKey: ['admin-users'] });
        },

        onSuccess: () => {
          toast.success('Acesso revogado com sucesso');
        },
      });
    }
    ```

    Key features:
    - Optimistic updates show immediate UI feedback
    - Rollback on error restores previous state
    - invalidateQueries triggers refetch after mutation
    - Toast notifications for success/error

    Reference 04-RESEARCH.md Pattern 4 for complete optimistic update pattern.

    import { useState } from 'react';
    import { Search, X } from 'lucide-react';
    import { Input } from '@/components/ui/input';
    import { useDebounce } from '@/hooks/useDebounce';

    interface UserSearchInputProps {
      value: string;
      onChange: (value: string) => void;
      placeholder?: string;
    }

    export function UserSearchInput({
      value,
      onChange,
      placeholder = 'Buscar por nome ou email...',
    }: UserSearchInputProps) {
      const [localValue, setLocalValue] = useState(value);
      const debouncedValue = useDebounce(localValue, 300);

      // Update parent with debounced value
      useState(() => {
        if (debouncedValue !== value) {
          onChange(debouncedValue);
        }
      });

      const handleChange = (newValue: string) => {
        setLocalValue(newValue);
      };

      const handleClear = () => {
        setLocalValue('');
        onChange('');
      };

      return (
        <div className="relative">
          <Search className="absolute left-3 top-1/2 -translate-y-1/2 text-muted-foreground" size={16} />
          <Input
            type="search"
            placeholder={placeholder}
            value={localValue}
            onChange={(e) => handleChange(e.target.value)}
            className="pl-10 pr-10"
          />
          {localValue && (
            <button
              onClick={handleClear}
              className="absolute right-3 top-1/2 -translate-y-1/2 text-muted-foreground hover:text-foreground"
              type="button"
            >
              <X size={16} />
            </button>
          )}
        </div>
      );
    }
    ```

    Key features:
    - Uses useDebounce hook with 300ms delay
    - Shows search icon on left, clear button on right
    - Controlled component with local state for immediate UI feedback
    - Clear button resets search

    import { Label } from '@/components/ui/label';
    import { Checkbox } from '@/components/ui/checkbox';
    import {
      Select,
      SelectContent,
      SelectItem,
      SelectTrigger,
      SelectValue,
    } from '@/components/ui/select';
import { Filter } from 'lucide-react';

    export type StatusFilter = 'active' | 'expired' | 'inactive';
    export type AccessTypeFilter = 'manual' | 'paid' | 'free';

    interface UserFiltersProps {
      statusFilters: StatusFilter[];
      onStatusChange: (filters: StatusFilter[]) => void;
      accessTypeFilters: AccessTypeFilter[];
      onAccessTypeChange: (filters: AccessTypeFilter[]) => void;
    }

    export function UserFilters({
      statusFilters,
      onStatusChange,
      accessTypeFilters,
      onAccessTypeChange,
    }: UserFiltersProps) {
      const handleStatusToggle = (status: StatusFilter) => {
        if (statusFilters.includes(status)) {
          onStatusChange(statusFilters.filter(s => s !== status));
        } else {
          onStatusChange([...statusFilters, status]);
        }
      };

      const handleAccessTypeToggle = (type: AccessTypeFilter) => {
        if (accessTypeFilters.includes(type)) {
          onAccessTypeChange(accessTypeFilters.filter(t => t !== type));
        } else {
          onAccessTypeChange([...accessTypeFilters, type]);
        }
      };

      const statusOptions: { value: StatusFilter; label: string }[] = [
        { value: 'active', label: 'Ativo' },
        { value: 'expired', label: 'Expirado' },
        { value: 'inactive', label: 'Inativo' },
      ];

      const accessTypeOptions: { value: AccessTypeFilter; label: string }[] = [
        { value: 'manual', label: 'Manual' },
        { value: 'paid', label: 'Pago' },
        { value: 'free', label: 'Grátis' },
      ];

      return (
        <div className="flex items-center gap-4">
          <div className="flex items-center gap-2">
            <Filter size={16} className="text-muted-foreground" />
            <Label className="text-sm">Filtros:</Label>
          </div>

          {/* Status Filter */}
          <Select>
            <SelectTrigger className="w-[180px]">
              <SelectValue placeholder="Status" />
            </SelectTrigger>
            <SelectContent>
              {statusOptions.map((option) => (
                <div key={option.value} className="flex items-center gap-2 px-2 py-1.5">
                  <Checkbox
                    checked={statusFilters.includes(option.value)}
                    onCheckedChange={() => handleStatusToggle(option.value)}
                    label={option.label}
                  />
                </div>
              ))}
            </SelectContent>
          </Select>

          {/* Access Type Filter */}
          <Select>
            <SelectTrigger className="w-[180px]">
              <SelectValue placeholder="Tipo de Acesso" />
            </SelectTrigger>
            <SelectContent>
              {accessTypeOptions.map((option) => (
                <div key={option.value} className="flex items-center gap-2 px-2 py-1.5">
                  <Checkbox
                    checked={accessTypeFilters.includes(option.value)}
                    onCheckedChange={() => handleAccessTypeToggle(option.value)}
                    label={option.label}
                  />
                </div>
              ))}
            </SelectContent>
          </Select>

          {/* Clear Filters Button */}
          {(statusFilters.length > 0 || accessTypeFilters.length > 0) && (
            <button
              onClick={() => {
                onStatusChange([]);
                onAccessTypeChange([]);
              }}
              className="text-sm text-muted-foreground hover:text-foreground"
            >
              Limpar filtros
            </button>
          )}
        </div>
      );
    }
    ```
