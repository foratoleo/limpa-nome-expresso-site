**Pattern:**
```typescript
import { describe, it, expect } from 'vitest';
import { supabaseAdmin, getPolicies } from './conftest';

describe('RLS Policies', () => {
  describe('user_access table security', () => {
    it('should block users from updating own access', async () => {
      const policies = await getPolicies('user_access');

      const updateDenyPolicy = policies.find(p =>
        p.policyname.includes('cannot update') &&
        p.cmd === 'UPDATE'
      );

      expect(updateDenyPolicy).toBeDefined();
      expect(updateDenyPolicy?.permissive).toBe(false); // USING (false) creates restrictive policy
    });

    it('should allow service role full access', async () => {
      const policies = await getPolicies('user_access');

      const serviceRolePolicy = policies.find(p =>
        p.roles.includes('service_role') &&
        p.cmd === 'ALL'
      );

      expect(serviceRolePolicy).toBeDefined();
    });
  });
});
```

**Pattern:**
```typescript
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { supabaseAdmin } from './conftest';

describe('Query Performance and Expiration', () => {
  describe('Expiration checking (DB-04)', () => {
    it('should exclude expired user_access records', async () => {
      const expiredDate = new Date(Date.now() - 86400000).toISOString(); // Yesterday

      const { data: inserted } = await supabaseAdmin
        .from('user_access')
        .insert({
          user_id: 'test-user-id',
          access_type: 'one_time',
          expires_at: expiredDate,
          is_active: true
        })
        .select()
        .single();

      const { data: activeAccess } = await supabaseAdmin
        .from('user_access')
        .select('*')
        .eq('user_id', 'test-user-id')
        .eq('is_active', true)
        .gte('expires_at', new Date().toISOString());

      expect(activeAccess).toBeNull();

      // Cleanup
      await supabaseAdmin.from('user_access').delete().eq('id', inserted.id);
    });
  });

  describe('Query performance (DB-01)', () => {
    it('should complete active access check in <100ms', async () => {
      const start = Date.now();
      const { data } = await supabaseAdmin
        .from('user_access')
        .select('*')
        .eq('user_id', 'test-user-id')
        .eq('is_active', true)
        .gte('expires_at', new Date().toISOString())
        .maybeSingle();
      const duration = Date.now() - start;

      expect(duration).toBeLessThan(100);
    });
  });
});
```
